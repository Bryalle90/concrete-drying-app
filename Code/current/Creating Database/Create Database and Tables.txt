CREATE DATABASE plasticcracks;

USE plasticcracks;

CREATE TABLE project (
	projectID int UNIQUE NOT NULL AUTO_INCREMENT,
	projectName varchar(40),
	location varchar(40),
	ownerID int NOT NULL,
	zipcode int,
	unit varchar(1),
	addedTime varchar(16),
	reminder varchar(16),
	PRIMARY KEY(projectID)
);

CREATE TABLE series (
	seriesID int UNIQUE NOT NULL AUTO_INCREMENT,
	projectID int,
	isOriginal varchar(1),
	concreteTemp double,
	windSpeed double,
	PRIMARY KEY(seriesID)
);

CREATE TABLE user (
	userID int UNIQUE NOT NULL AUTO_INCREMENT,
	name varchar(40),
	email varchar(254),
	userPass varchar(80),
	code varchar(60),
	forgotCode varchar(60),
	createdTime varchar(16),
	isAdmin char(1),
	isValidated tinyint(1),
	seenNotifMsg tinyint(1),
	forceNewPass tinyint(1),
	PRIMARY KEY(userID)
);

CREATE TABLE userProjectLookup (
	userID int NOT NULL,
	projectID int NOT NULL,
	accepted tinyint(1),
	PRIMARY KEY(userID, projectID)
);

CREATE TABLE weather (
	zipcode int NOT NULL,
	evapRate double,
	cloudCoverage double,
	airTemp double,
	concTemp double,
	humidity double,
	windSpeed double,
	weatherTime varchar(16),
	PRIMARY KEY(zipcode, weatherTime)
);

CREATE TABLE zipUpdate (
	zipcode int NOT NULL,
	updateTime varchar(16),
	PRIMARY KEY(zipcode)
);

CREATE TABLE changeInStateNotification (
	changeInStateNotificationID int UNIQUE NOT NULL AUTO_INCREMENT,	
	projectID int,
	seriesID int,
	time varchar(16),
	currentZone tinyint(1),
	notifyZone tinyint(1),
	createdDate varchar(16),
	PRIMARY KEY(changeInStateNotificationID)
);

CREATE TABLE changeInStateNotificationLog (
	changeInStateNotificationID int,
	dateSent varchar(16),
	PRIMARY KEY(changeInStateNotificationID)
);

CREATE TABLE futureNotificationLog (
	futureID int,
	dateSent varchar(16),
	PRIMARY KEY(futureID)
);

CREATE TABLE zipcodeLog (
	zipcode int,
	userType varchar(1),
	date varchar(16),
	PRIMARY KEY(zipcode, userType, date)
);

CREATE TABLE zipinfo (
	zipcode int NOT NULL,
	city varchar(40),
	state varchar(40),
	latitude float,
	longitude float,
	PRIMARY KEY(zipcode)
);
